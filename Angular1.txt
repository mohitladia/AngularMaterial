 @ Component({
    selector:'courses',
    template:`

            <li *ngFor="let course of courses">
                {{course}}
            </li>
        </ul>`
 })
 export class CourseComponent{
     title = "List of Courses";
     getTitle(){
         return this.title;
     }
     courses =["course1","course2","course3"]
 }

 }

}
